#include <muu/static_string.h>
template<typename Character, size_t Length>
muu::static_string class

A null-terminated string for compile-time string manipulation.

Template parameters
Character The character type of the string.
Length The length of the string (not including the null terminator).

template<typename Character, size_t Length>
template<char... Str>
static_string<char, sizeof...(Str)> operator""_ss() constexpr

Constructs a static_string exactly representing a numeric literal.

constexpr auto str = 3.141_ss;
static_assert(str == static_string{ "3.141" })

template<typename Character, size_t Length>
template<impl::static_string_udl_impl Str>
auto operator""_ss() constexpr

Constructs a static_string directly using a string literal.

constexpr auto str = "hello"_ss;
static_assert(str == static_string{ "hello" })

Conditional availability

This operator depends on C++20's class-type NTTPs and three-way comparison operator. Check for support using MUU_HAS_STATIC_STRING_LITERALS and/or build::supports_static_string_literals.
